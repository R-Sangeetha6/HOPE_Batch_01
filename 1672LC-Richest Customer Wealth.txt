class Solution {
    public int maximumWealth(int[][] accounts) {
        int max=Integer.MIN_VALUE;
        int wealth=0;
    for (int[] matrix:accounts){
        for(int i=0;i<matrix.length;i++){
            wealth+=matrix[i];

        }
        if(max<wealth)
        max=wealth;
    
      
             wealth=0;
    }
    return max;
    }
}
