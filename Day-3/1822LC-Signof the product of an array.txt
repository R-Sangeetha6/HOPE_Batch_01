class Solution {
    public int arraySign(int[] nums) {
        int value=1;
        for(int i=0;i<nums.length;i++){
            value*=nums[i];
        }
        return value;
    }
     public  int signFunc(int i) {
        if(i==0){
            return 0;
        }
        if(i>0){
            return 1;

        }
        else{
            return -1;
        }
    
        
    }
}
